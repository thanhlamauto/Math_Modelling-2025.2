\begin{frame}{Roadmap}
  \begin{enumerate}
    \item Problem and modeling approach
    \item Spatial SIR model with two superspreader mechanisms
    \item Monte Carlo solving algorithm
    \item Reproduced simulation results
    \item Sensitivity, robustness, and reference conclusion
  \end{enumerate}
\end{frame}

\begin{frame}{Problem}
  \begin{columns}[T,totalwidth=\linewidth]
    \begin{column}{0.55\linewidth}
      \begin{itemize}
        \item Classical well-mixed SIR models assume similar transmission rates for all infected individuals.
        \item Real outbreaks often show strong heterogeneity: a small number of cases cause many secondary infections.
        \item The paper models superspreaders directly and asks which mechanism explains spread better.
      \end{itemize}
    \end{column}
    \begin{column}{0.40\linewidth}
      \begin{block}{Key question}
        Are superspreaders important because they are more infectious locally, or because they have many social connections?
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Modeling Approach}
  \begin{itemize}
    \item Individuals are fixed in a two-dimensional continuous domain of side length $L=10r_0$.
    \item Each individual has state $S$, $I$, or $R$.
    \item Transmission probability depends on periodic distance $r$ between an infected individual and a susceptible individual.
    \item A fraction $\lambda$ of the population is assigned as superspreaders.
  \end{itemize}
  \vspace{0.3cm}
  \[
    \rho=\frac{N}{L^2},\qquad X_i(t)\in\{S,I,R\}.
  \]
\end{frame}

\begin{frame}{Two Superspreader Models}
  \begin{columns}[T,totalwidth=\linewidth]
    \begin{column}{0.49\linewidth}
      \begin{block}{Strong infectiousness model}
        Superspreaders have higher local infection probability but the same cutoff radius:
        \[
        w_s(r)=
        \begin{cases}
          w_0,&0\le r\le r_0,\\
          0,&r>r_0.
        \end{cases}
        \]
      \end{block}
    \end{column}
    \begin{column}{0.49\linewidth}
      \begin{block}{Hub model}
        Superspreaders have a wider contact radius:
        \[
        w_s(r)=
        \begin{cases}
          w_0(1-r/r_n)^2,&0\le r\le r_n,\\
          0,&r>r_n,
        \end{cases}
        \]
        with $r_n=\sqrt{6}r_0$ for superspreaders.
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.2cm}
  \smallnote{Both mechanisms are normalized to have comparable expected infection strength; their spatial patterns differ.}
\end{frame}

\begin{frame}{Basic Reproductive Number and Critical Density}
  \[
  R_0(\lambda)=
  \frac{\rho}{w_0}
  \left[
    \lambda \int w_s(r)2\pi r\,dr+
    (1-\lambda)\int w_n(r)2\pi r\,dr
  \right].
  \]
  \[
    \rho_c\pi r_0^2=
    \frac{R_c}{\lambda+\frac{1-\lambda}{6}}.
  \]
  \begin{columns}[T,totalwidth=\linewidth]
    \begin{column}{0.48\linewidth}
      \begin{block}{Strong infectiousness}
        $R_c \approx 4.5$
      \end{block}
    \end{column}
    \begin{column}{0.48\linewidth}
      \begin{block}{Hub model}
        $R_c \approx 3.2$
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Monte Carlo Solving Algorithm}
  \begin{enumerate}
    \item Randomly place $N$ individuals in the domain; set one initial infected individual.
    \item Randomly assign superspreader labels using fraction $\lambda$.
    \item At each step, infected individuals attempt to infect susceptible individuals with probability $w(r)$.
    \item Newly infected individuals transmit only from the next step.
    \item Current infected individuals recover with probability $\gamma$.
    \item Stop when no infected individuals remain or the maximum number of steps is reached.
  \end{enumerate}
  \vspace{0.2cm}
  \smallnote{Outputs: percolation probability, front distance, propagation velocity, epidemic curve, and secondary infection distribution.}
\end{frame}

\begin{frame}{Comparison with SARS Data}
  \begin{itemize}
    \item The reference paper compares simulations with SARS in Singapore in 2003.
    \item The observed direct secondary cases include several unusually large values, including 12, 21, 23, and 40.
    \item The hub model with $N=477$, $\rho\pi r_0^2=15.0$, $\lambda=0.4$, and $\gamma=1$ gives the best qualitative epidemic-curve match.
  \end{itemize}
  \begin{block}{Reference conclusion}
    SARS superspreading was better explained by many social connections than by stronger local infectiousness alone.
  \end{block}
\end{frame}

\begin{frame}{Sensitivity and Robustness}
  \begin{itemize}
    \item The model is sensitive to superspreader fraction $\lambda$: increasing $\lambda$ lowers critical density and increases propagation speed.
    \item It is sensitive to the mechanism of superspreading: hub contacts produce faster spatial spread than stronger local infectiousness.
    \item Population density controls whether infection chains die out or percolate.
    \item Recovery probability $\gamma$ is also important: lower $\gamma$ means a longer infectious period and more transmission opportunities.
  \end{itemize}
  \begin{block}{Robust qualitative findings}
    Across parameter settings, superspreaders lower the percolation threshold, speed up spread, sharpen epidemic curves, and create long-tailed secondary-infection distributions.
  \end{block}
\end{frame}

\begin{frame}{Limitations}
  \begin{itemize}
    \item Individuals are fixed in space, while real people move and change contacts.
    \item Superspreader labels are assigned randomly, but real superspreading depends on behavior, occupation, environment, and biology.
    \item The model is best interpreted as a controlled comparison of two mechanisms, not a full prediction tool for every epidemic.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusion}
  \begin{enumerate}
    \item Superspreaders substantially change epidemic dynamics.
    \item The hub model spreads faster and has lower critical density than the strong infectiousness model.
    \item Long-range social connectivity is the key mechanism in the reference paper's SARS comparison.
  \end{enumerate}
  \vspace{0.35cm}
  \begin{block}{Takeaway}
    Average transmission rate alone is not enough; epidemic models should include heterogeneity in individual connectivity.
  \end{block}
\end{frame}
